## Lesson 6 Print 9*9 multiplication table 乘法表
### 代码
	#include <stdio.h>

	int main(void)
	{		
		int i, j;

		for (i = 1; i < 10; i++)
		{	
			for (j = 1; j <= i; j++)
			{
				printf("%d*%d=%d\t", j, i, i * j);
			}
			printf("\n");
		}

		return 0;
	}
